211 public FilterBuilder AddManufacturerData(ushort companyId, byte[] manufacturerData)
212 {
213 //create a manufacturer data section we wanted to match for
214 var _manufacturerData = new BluetoothLEManufacturerData();
215
216 // Then, set the company ID for the manufacturer data. (For testing: 0xFFFE)
217 _manufacturerData.CompanyId = companyId;
218
219 // Finally set the data payload within the manufacturer-specific section
220 // Here, use a 16-bit UUID: 0x1234 -> {0x34, 0x12} (little-endian)
221 var writer = new DataWriter();
222 writer.WriteBytes(manufacturerData);
223
224 // Make sure that the buffer length can fit within an advertisement payload (20 bytes). Otherwise you will get an exception.
225 _manufacturerData.Data = writer.DetachBuffer();
226
227 // Add the manufacturer data to the advertisement filter on the watcher:
228 filter.AdvertisementFilter.Advertisement.ManufacturerData.Add(_manufacturerData);
229
230 return this;
231 }

244 public FilterBuilder SetDataBuffer(AdvertisementSectionType dataType, byte[] data)
245 {

247 try

249 dataSection.DataType = ((byte)dataType);
250 var writer = new DataWriter();

366 public CosmedBluetoothLEAdvertisementFilter SetSignalStrengthFilter(short InRangeThresholdInDBm, short OutOfRangeThresholdInDBm, TimeSpan OutOfRangeTimeout)
367 {
368 //Set the in-range threshold to -70dBm. This means advertisements with RSSI >= -70dBm
369 //will start to be considered "in-range"
370 SignalStrengthFilter.InRangeThresholdInDBm = InRangeThresholdInDBm;
371
372 // Set the out-of-range threshold to -75dBm (give some buffer). Used in conjunction with OutOfRangeTimeout
373 // to determine when an advertisement is no longer considered "in-range"
374 SignalStrengthFilter.OutOfRangeThresholdInDBm = OutOfRangeThresholdInDBm;
375
376 // Set the out-of-range timeout to be 2 seconds. Used in conjunction with OutOfRangeThresholdInDBm
377 // to determine when an advertisement is no longer considered "in-range"
378 SignalStrengthFilter.OutOfRangeTimeout = OutOfRangeTimeout;
379
380 return this;
381 }
