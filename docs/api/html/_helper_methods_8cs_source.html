272 public static string ConvertValueBufferToString(IBuffer value, GattPresentationFormat format = null)
273 {
274 // no format, return bytes
275 if (format == null)
276 {
277 return ClientBufferReader.ToHexString(value);
278 }
279
280 // Bool
281 if (format.FormatType == GattPresentationFormatTypes.Boolean)
282 {
283 // Previous implementation was incorrect. Need to implement in GattHelper.
284 throw new NotImplementedException();
285 }
286 else if (format.FormatType == GattPresentationFormatTypes.Bit2 ||
287 format.FormatType == GattPresentationFormatTypes.Nibble)
288 {
289 // 2bit or nibble - no exponent
290 // Previous implementation was incorrect. Need to implement in GattHelper.
291 return ClientBufferReader.ToHexString(value);
292 }
293 else if (format.FormatType == GattPresentationFormatTypes.UInt8 ||
294 format.FormatType == GattPresentationFormatTypes.UInt12 ||
295 format.FormatType == GattPresentationFormatTypes.UInt16)
296 {
297 // Previous implementation was incorrect. Need to implement in GattHelper.
298 return ClientBufferReader.ToHexString(value);
299 }
300 else if (format.FormatType == GattPresentationFormatTypes.UInt24 ||
301 format.FormatType == GattPresentationFormatTypes.UInt32)
302 {
303 // Previous implementation was incorrect. Need to implement in GattHelper.
304 return ClientBufferReader.ToHexString(value);
305 }
306 else if (format.FormatType == GattPresentationFormatTypes.UInt48 ||
307 format.FormatType == GattPresentationFormatTypes.UInt64)
308 {
309 // Previous implementation was incorrect. Need to implement in GattHelper.
310 return ClientBufferReader.ToHexString(value);
311 }
312 else if (format.FormatType == GattPresentationFormatTypes.SInt8 ||
313 format.FormatType == GattPresentationFormatTypes.SInt12 ||
314 format.FormatType == GattPresentationFormatTypes.SInt16)
315 {
316 // Previous implementation was incorrect. Need to implement in GattHelper.
317 return ClientBufferReader.ToHexString(value);
318 }
319 else if (format.FormatType == GattPresentationFormatTypes.SInt24 ||
320 format.FormatType == GattPresentationFormatTypes.SInt32)
321 {
322 return ClientBufferReader.ToInt32(value).ToString();
323 }
324 else if (format.FormatType == GattPresentationFormatTypes.Utf8)
325 {
326 return ClientBufferReader.ToUTF8String(value);
327 }
328 else if (format.FormatType == GattPresentationFormatTypes.Utf16)
329 {
330 return ClientBufferReader.ToUTF16String(value);
331 }
332 else
333 {
334 // format.FormatType == GattPresentationFormatTypes.UInt128 ||
335 // format.FormatType == GattPresentationFormatTypes.SInt128 ||
336 // format.FormatType == GattPresentationFormatTypes.DUInt16 ||
337 // format.FormatType == GattPresentationFormatTypes.SInt64 ||
338 // format.FormatType == GattPresentationFormatTypes.Struct ||
339 // format.FormatType == GattPresentationFormatTypes.Float ||
340 // format.FormatType == GattPresentationFormatTypes.Float32 ||
341 // format.FormatType == GattPresentationFormatTypes.Float64
342 return ClientBufferReader.ToHexString(value);
343 }
344 }
