74 protected string convertBufferData(IBuffer buffer, DataConversionType type)
75 {
76 var data = new byte[buffer.Length];
77 using (var reader = DataReader.FromBuffer(buffer))
78 {
79 reader.ReadBytes(data);
80 }
81
82 string result;
83
84 switch (type)
85 {
86 case DataConversionType.Hex:
87 result = BitConverter.ToString(data);
88 break;
89
90 case DataConversionType.ASCII:
91 result = Encoding.ASCII.GetString(data);
92 break;
93
94 case DataConversionType.Utf8:
95 result = Encoding.UTF8.GetString(data);
96 break;
97
98 case DataConversionType.Utf16:
99 result = Encoding.Unicode.GetString(data);
100 break;
101
102 default:
103 result = "";
104 break;
105 }
106 return result;
107
108 }

242 public static IBuffer ConvertValueToBuffer(DateTime time)
243 {
244 DataWriter writer = new DataWriter();
245
246 // Date time according to: https://www.bluetooth.com/specifications/gatt/viewer?attributeXmlFile=org.bluetooth.characteristic.date_time.xml
247 writer.WriteUInt16((ushort)time.Year);
248 writer.WriteByte((byte)time.Month);
249 writer.WriteByte((byte)time.Day);
250 writer.WriteByte((byte)time.Hour);
251 writer.WriteByte((byte)time.Minute);
252 writer.WriteByte((byte)time.Second);
253
254 // Day of week according to: https://www.bluetooth.com/specifications/gatt/viewer?attributeXmlFile=org.bluetooth.characteristic.day_of_week.xml
255 // Going to leave this "not known" for now - would have to perform a rotate of DayOfWeek property
256 writer.WriteByte(0x0);
257
258 return writer.DetachBuffer();
259 }

267 public static IBuffer ToIBufferFromHexString(string data)
268 {
269 DataWriter writer = new DataWriter();
270 data = data.Replace("-", "");
271
272 if (data.Length > 0)
273 {
274 if (data.Length % 2 != 0)
275 {
276 data = "0" + data;
277 }
278
279 int NumberChars = data.Length;
280 byte[] bytes = new byte[NumberChars / 2];
281
282 for (int i = 0; i < NumberChars; i += 2)
283 {
284 bytes[i / 2] = Convert.ToByte(data.Substring(i, 2), 16);
285 }
286 writer.WriteBytes(bytes);
287 }
288 return writer.DetachBuffer();
289 }

495 reader.ByteOrder = Windows.Storage.Streams.ByteOrder.LittleEndian;

677 private static byte[] PadBytes(byte[] input, int length)
678 {
679 if (input.Length >= length)
680 {
681 return input;
682 }
683
684 byte[] ret = new byte[length];
685 Array.Copy(input, ret, input.Length);
686 return ret;
687 }
