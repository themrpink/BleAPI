276 private async void scanInit()
277 {
278 clearDiscoveredDevices(discoveredDevices);
279 clearDiscoveredDevices(lastDiscoveredDevices);
280
281 if (watcher == null)
282 {
283 if (!IsFilteringActive)
284 {
285 watcher = new BluetoothLEAdvertisementWatcher();
286 }
287 else
288 {
289 if (filter.AdvertisementFilter != null)
290 {
291 watcher = new BluetoothLEAdvertisementWatcher(filter.AdvertisementFilter);
292 }
293 else
294 {
295 watcher = new BluetoothLEAdvertisementWatcher();
296 }
297 if (filter.SignalStrengthFilter != null)
298 {
299 watcher.SignalStrengthFilter = filter.SignalStrengthFilter;
300 }
301
302 }
303
304 watcher.Received += this.OnAdvertisementReceived;
305 watcher.Stopped += this.OnScanStopped;
306 checkFilterStatus();
307 }
308
309 //Add the extended advertising, if supported
310 var adapter = await CosmedBluetoothLEAdapter.CreateAsync();
311 bool isExtendedAdvertisementSupported = adapter.IsExtendedAdvertisingSupported;
312 watcher.AllowExtendedAdvertisements = isExtendedAdvertisementSupported;
313 }

320 private async Task scan()
321 {
322 try
323 {
324 if (status == StateMachine.Starting && await checkBLEAdapterError())
325 {
326 if (watcher != null)
327 {
328 lastScanningMode = watcher.ScanningMode;
329 watcher.Start();
330 status = StateMachine.Started;
331
332 int count = 0;
333 while (watcher.Status != BluetoothLEAdvertisementWatcherStatus.Started && count < wait)
334 {
335 count++;
336 }
337 }
338 }
339
340 StartedListening?.Invoke(this, watcher.ScanningMode);
341 }
342 catch (System.Exception e)
343 {
344 status = StateMachine.Aborted;
345 await Task.Run(() => ScanInterrupted?.Invoke(this, new ScanAbortedException("scan aborted during initiation", e))).ConfigureAwait(false);
346 }
347 }

353 public void StopScanning()
354 {
355 if (status == StateMachine.Started || status == StateMachine.Paused)
356 {
357 status = StateMachine.Stopping;
358
359 //the handler is removed to stop the send of pending discovered devices
360 watcher.Received -= OnAdvertisementReceived;
361 watcher.Stop();
362 status = StateMachine.Stopped;
363
364 int count = 0;
365
366 //add a small delay to allow the background scan to effectively stop and avoid an anticipated
367 //change of state
368 while (watcher.Status != BluetoothLEAdvertisementWatcherStatus.Stopped && count < wait)
369 {
370 count++;
371 }
372
373 //watcher.Stopped -= OnScanStopped;
374 watcher = null;
375 }

543 private async Task<bool> checkBLEAdapterError()
544 {
545 try
546 {
547 CosmedBluetoothLEAdapter adapter = await CosmedBluetoothLEAdapter.CreateAsync();
548 string str = adapter.HexAddress;
549 Console.WriteLine(str);
550
551 if (!adapter.IsLowEnergySupported)
552 {
553 status = StateMachine.Aborted;
554 var e = new BluetoothLeNotSupportedException("Your adapter " + str + " does not support Bluetooth Low Energy");
555 await Task.Run(() => ScanInterrupted?.Invoke(this, e)).ConfigureAwait(false);
556 return false;
557 }
558 if (!adapter.IsCentralRoleSupported)
559 {
560 status = StateMachine.Aborted;
561 var e = new CentralRoleNotSupportedException("Your adapter does not support Central Role");
562 await Task.Run(() => ScanInterrupted?.Invoke(this, e)).ConfigureAwait(false);
563 return false;
564 }
565 }
566 catch (BluetoothAdapterCommunicationFailureException e)
567 {
568 status = StateMachine.Aborted;
569 await Task.Run(() => ScanInterrupted?.Invoke(this, e)).ConfigureAwait(false);
570 return false;
571 }
572
573 return true;
574

656 private void OnAdvertisementReceived(BluetoothLEAdvertisementWatcher sender, BluetoothLEAdvertisementReceivedEventArgs args)
657 {
658 ICosmedBleAdvertisedDevice device;
659 Console.WriteLine("adv received +++++++++++++++");
660 if (sender == watcher && args != null && status == StateMachine.Started)
661 {
662 lock (devicesThreadLock)
663 {
664 bool newDevice = !discoveredDevices.ContainsKey(args.BluetoothAddress);
665 if (newDevice)
666 {
667 if (discoveredDevices.Count >= maxScanResults)
668 {
669 CleanOlderDiscoveredDevices(discoveredDevices);
670 }
671 device = new CosmedBleAdvertisedDevice(args);
672 discoveredDevices[args.BluetoothAddress] = device;
673 }
674 else
675 {
676 discoveredDevices[args.BluetoothAddress].SetAdvertisement(args);
677 }
678
679 if (lastDiscoveredDevices.Count >= maxScanResults)
680 {
681 CleanOlderDiscoveredDevices(lastDiscoveredDevices);
682 }
683
684 lastDiscoveredDevices[args.BluetoothAddress] = discoveredDevices[args.BluetoothAddress];
685 device = discoveredDevices[args.BluetoothAddress];
686 }
687 Task.Run(() => { NewDeviceDiscovered?.Invoke(this, device); }).ConfigureAwait(false);
688 }

699 private void OnScanStopped(BluetoothLEAdvertisementWatcher sender, BluetoothLEAdvertisementWatcherStoppedEventArgs args)
700 {
701 if (sender.Status == BluetoothLEAdvertisementWatcherStatus.Aborted)
702 {
703 status = StateMachine.Aborted;
704 Task.Run(() => ScanInterrupted?.Invoke(this, new ScanAbortedException("scan aborted, please check your Bluetooth adapter"))).ConfigureAwait(false);
705 }
706 else
707 {
708 Task.Run(() => ScanStopped?.Invoke(this, args)).ConfigureAwait(false);
709 Console.WriteLine("Stopping scan");
710 Console.WriteLine(args.Error.ToString());
711 }
