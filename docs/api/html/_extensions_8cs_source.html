123 public static async Task<CosmedCharacteristicWriteResult> WriteWithResult(this GattCharacteristic characteristic, byte[] value, GattWriteOption writeOption, Action<GattCharacteristic, CosmedGattErrorFoundEventArgs> errorAction = null)
124 {
125 GattCharacteristicProperties properties = characteristic.CharacteristicProperties;
126
127 if(writeOption == GattWriteOption.WriteWithoutResponse)
128 {
129 properties = GattCharacteristicProperties.WriteWithoutResponse;
130 }
131 else if(writeOption == GattWriteOption.WriteWithResponse)
132 {
133 properties = GattCharacteristicProperties.Write;
134 }
135
136 if (characteristic.IsWriteAllowed())
137 {
138 //check MaxPduSize from GattSession before use
139 var writer = new DataWriter();
140 // WriteByte used for simplicity. Other common functions - WriteInt16 and WriteSingle
141
142 try
143 {
144 writer.WriteBytes(value);
145
146 if (characteristic.CharacteristicProperties.HasFlag(GattCharacteristicProperties.Write))
147 {
148 var statusResultValue = await characteristic.WriteValueWithResultAsync(writer.DetachBuffer(), writeOption).AsTask().ConfigureAwait(false);
149 return new CosmedCharacteristicWriteResult(statusResultValue.ProtocolError, statusResultValue.Status.ConvertStatus());
150 }
151
152 // write without response cannot write values larger than MTU as per spec. Any longer writes can only be handled with response.
153 else if (characteristic.CharacteristicProperties.HasFlag(GattCharacteristicProperties.WriteWithoutResponse))
154 {
155 var statusResultValue = await characteristic.WriteValueWithResultAsync(writer.DetachBuffer(), GattWriteOption.WriteWithoutResponse).AsTask().ConfigureAwait(false);
156 return new CosmedCharacteristicWriteResult(statusResultValue.ProtocolError, statusResultValue.Status.ConvertStatus());
157 }
158
159 writer.Dispose();
160 }
161 catch (Exception e)
162 {
163 var args = new CosmedGattErrorFoundEventArgs(e, properties);
164 if (errorAction != null)
165 {
166 GattCharacteristicExtensions.ErrorFoundCustom += errorAction;
167 GattCharacteristicExtensions.ErrorFoundCustom.Invoke(characteristic, args);
168 GattCharacteristicExtensions.ErrorFoundCustom -= errorAction;
169 }
170 else
171 {
172 ErrorFoundClass.Call(characteristic, args);
173 }
174
175 //Console.WriteLine("error catched with characteristic: " + characteristic.Uuid.ToString());
176 //Console.WriteLine(e.Message);
177
178 return new CosmedCharacteristicWriteResult(null, CosmedGattCommunicationStatus.Unreachable);
179 }
180 }
181 return new CosmedCharacteristicWriteResult(null, CosmedGattCommunicationStatus.OperationNotSupported);
182 }

288 public static async Task<CosmedGattCommunicationStatus> SubscribeToNotification(this GattCharacteristic characteristic, TypedEventHandler<GattCharacteristic, GattValueChangedEventArgs> valueChangedAction, Action<GattCharacteristic, CosmedGattErrorFoundEventArgs> errorAction = null)
289 {
290 if (!characteristic.IsNotificationAllowed() && !characteristic.IsIndicationAllowed())
291 {
292 return CosmedGattCommunicationStatus.OperationNotSupported;
293 }
294
295 GattReadClientCharacteristicConfigurationDescriptorResult cccd;
296
297 try
298 {
299 Console.WriteLine("characteristec UUID: " + characteristic.Uuid.ToString());
300 cccd = await characteristic.ReadClientCharacteristicConfigurationDescriptorAsync();
301
302 if (cccd.Status != GattCommunicationStatus.Success)
303 {
304 return cccd.Status.ConvertStatus();
305 }
306
307 if (cccd.ClientCharacteristicConfigurationDescriptor == GattClientCharacteristicConfigurationDescriptorValue.None)
308 {
309 return CosmedGattCommunicationStatus.OperationNotRegistered;
310 }
311
312 GattCommunicationStatus status = await characteristic.WriteClientCharacteristicConfigurationDescriptorAsync(GattClientCharacteristicConfigurationDescriptorValue.Notify).AsTask().ConfigureAwait(false);
313
314 if (status == GattCommunicationStatus.Success)
315 {
316 if (valueChangedAction != null)
317 {
318 characteristic.ValueChanged += valueChangedAction;
319 if (GattCharacteristicEventsCollector.CharacteristicsChangedSubscriptions.ContainsKey(characteristic))
320 {
321 TypedEventHandler<GattCharacteristic, GattValueChangedEventArgs> oldEventHandler;
322 var success = GattCharacteristicEventsCollector.CharacteristicsChangedSubscriptions.TryGetValue(characteristic, out oldEventHandler);
323 if (success)
324 {
325 characteristic.ValueChanged -= oldEventHandler;
326 }
327 }
328 GattCharacteristicEventsCollector.CharacteristicsChangedSubscriptions[characteristic] = valueChangedAction;
329 }
330 else
331 {
332 throw new ArgumentNullException("valueChangedAction parameter cannot be null");
333 }
334 }
335
336 //gives the last status
337 return status.ConvertStatus();
338
339 }
340 catch (Exception e)
341 {
342 var args = new CosmedGattErrorFoundEventArgs(e, GattCharacteristicProperties.Notify);
343 if (errorAction != null)
344 {
345 GattCharacteristicExtensions.ErrorFoundCustom += errorAction;
346 GattCharacteristicExtensions.ErrorFoundCustom.Invoke(characteristic, args);
347 GattCharacteristicExtensions.ErrorFoundCustom -= errorAction;
348 }
349 else
350 {
351 ErrorFoundClass.Call(characteristic, args);
352 }
353
354 Console.WriteLine("error catched with characteristic: " + characteristic.Uuid.ToString());
355 Console.WriteLine(e.Message);

357 return CosmedGattCommunicationStatus.Unreachable;
358 }

578 public static async Task<BluetoothAppearanceType> GetAppearanceValue(this GattCharacteristic characteristic)
579 {
580 var result = await characteristic.Read();
581
582 var data = new byte[result.RawData.Length];
583 using (var reader = DataReader.FromBuffer(result.RawData))
584 {
585 reader.ReadBytes(data);
586 }
587 if(data.Length > 0)
588 {
589 ushort dataType = BitConverter.ToUInt16(data, 0);
590
591 BluetoothAppearanceType appearenceType;
592 if (Enum.TryParse(dataType.ToString(), out appearenceType))
593 {
594 return appearenceType;
595 }
596 }
597
598
599 return BluetoothAppearanceType.Unknown;
600 }

772 public static Task<TResult> ToTask<TResult>(this IAsyncOperation<TResult> asyncOperation)
773 {
774 var tsc = new TaskCompletionSource<TResult>();
775
776 asyncOperation.Completed += delegate
777 {
778 switch (asyncOperation.Status)
779 {
780 case AsyncStatus.Completed:
781 tsc.TrySetResult(asyncOperation.GetResults());
782 break;
783 case AsyncStatus.Error:
784 tsc.TrySetException(asyncOperation.ErrorCode);
785 break;
786 case AsyncStatus.Canceled:
787 tsc.SetCanceled();
788 break;
789 }
790 };
791 return tsc.Task;
