260 public CosmedBleAdvertisedDevice SetAdvertisement(BluetoothLEAdvertisementReceivedEventArgs args)
261 {
262 if (args == null)
263 throw new ArgumentNullException();
264
265 DeviceAddress = args.BluetoothAddress;
266 Timestamp = args.Timestamp;
267 RawSignalStrengthInDBm = args.RawSignalStrengthInDBm;
268 BluetoothAddressType = args.BluetoothAddressType;
269 IsAnonymous = args.IsAnonymous;
270 IsDirected = args.IsDirected;
271 IsScannable = args.IsScannable;
272 TransmitPowerLevelInDBm = args.TransmitPowerLevelInDBm;
273
274 //if the advertisement is a scan response saves it as such
275 if (args.AdvertisementType == BluetoothLEAdvertisementType.ScanResponse)
276 {
277 if (DeviceName == null || !args.Advertisement.LocalName.Equals(""))
278 {
279 DeviceName = args.Advertisement.LocalName;
280 }
281
282 scanResponseAdvertisementContent.Advertisement = args.Advertisement;
283 scanResponseAdvertisementContent.AdvertisementType = args.AdvertisementType;
284 HasScanResponse = true;
285
286 //rise the event
287 ScanResponseReceived?.Invoke(this);
288 }
289 //if the advertisement is normal advertisement saves it as such
290 else
291 {
292 IsConnectable = args.IsConnectable;
293 DeviceName = args.Advertisement.LocalName;
294 advertisementContent.Advertisement = args.Advertisement;
295 advertisementContent.AdvertisementType = args.AdvertisementType;
296 }
297 return this;
298 }
