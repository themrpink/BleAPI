175 try
176 {
177 // Verify: BT_Code: BluetoothLEDevice.FromIdAsync must be called from a UI thread because it may prompt for consent.

179 this.bluetoothLeDevice = await task.AsTask().ConfigureAwait(false);
180 }
181 catch (Exception e)
182 {
183 throw new BleDeviceConnectionException("Impossible to connect to device", e);
184 }
185

188 {
189 throw new BleDeviceConnectionException("Impossible to connect to device: device " + deviceAddress + " is null");
190 }
191

194 bluetoothLeDevice.ConnectionStatusChanged += ConnectionStatusChangedHandler;

298 public void ClearBluetoothLEDevice()
299 {
300 // Need to clear the CCCD from the remote device to stop receiving notifications
301 List<Task<GattCharacteristic>> removals = GattCharacteristicEventsCollector.CharacteristicsChangedSubscriptions.Where(c => c.Key.Service.Session.DeviceId.Equals(bluetoothLeDevice.DeviceId)).Select(async d =>
302 {
303 var result = await d.Key.WriteClientCharacteristicConfigurationDescriptorAsync(GattClientCharacteristicConfigurationDescriptorValue.None);
304 if (result == GattCommunicationStatus.Success)
305 {
306 d.Key.ValueChanged -= d.Value;
307 }
308 return d.Key;
309
310 }).ToList();
311
312 TypedEventHandler<GattCharacteristic, GattValueChangedEventArgs> eventHandler;
313 foreach (var r in removals)
314 {
315 var result = GattCharacteristicEventsCollector.CharacteristicsChangedSubscriptions.TryRemove(r.Result, out eventHandler);
316 }
317
318 bluetoothLeDevice.ConnectionStatusChanged -= ConnectionStatusChangedHandler;
319 bluetoothLeDevice.NameChanged -= NameChangedHandler;
320 bluetoothLeDevice.DeviceAccessInformation.AccessChanged -= AccessChangedHanlder;
321
322 bluetoothLeDevice?.Dispose();
323 bluetoothLeDevice = null;
324
325 GC.Collect();
326 }

398 //this method, used as default handler to manage the pairing process should be implemented by the user and passed
399 //to the GetPairedDevice method that accept an event as argument
400 private static void PairingRequestedHandler(DeviceInformationCustomPairing sender, DevicePairingRequestedEventArgs args)
401 {
402 switch (args.PairingKind)
403 {
404 case DevicePairingKinds.ConfirmOnly:
405 // Windows itself will pop the confirmation dialog as part of "consent" if this is running on Desktop or Mobile
406 // If this is an App for 'Windows IoT Core' where there is no Windows Consent UX, you may want to provide your own confirmation.
407 Console.WriteLine("ok presse enter to accept pairing");
408 Console.ReadLine();
409 args.Accept();
410 break;
411
412 case DevicePairingKinds.DisplayPin:
413 // We just show the PIN on this side. The ceremony is actually completed when the user enters the PIN
414 // on the target device. We automatically accept here since we can't really "cancel" the operation
415 // from this side.
416 Console.WriteLine("ok presse enter to accept pin " + args.Pin);
417 Console.ReadLine();
418 args.Accept();
419
420 // No need for a deferral since we don't need any decision from the user
421 Console.WriteLine("Please enter this PIN on the device you are pairing with: " + args.Pin, args.PairingKind);
422 Console.ReadLine();
423 break;
424
425 case DevicePairingKinds.ProvidePin:
426 // A PIN may be shown on the target device and the user needs to enter the matching PIN on
427 // this Windows device. Get a deferral so we can perform the async request to the user.
428 var collectPinDeferral = args.GetDeferral();
429
430 Console.WriteLine("Please enter the PIN shown on the device you're pairing with");
431 string pin = Console.ReadLine();
432 if (!string.IsNullOrEmpty(pin))
433 {
434 args.Accept(pin);
435 }
436
437 collectPinDeferral.Complete();
438
439 break;
440
441 case DevicePairingKinds.ProvidePasswordCredential:
442
443 var collectCredentialDeferral = args.GetDeferral();
444 Console.WriteLine("insert username");
445 string username = Console.ReadLine();
446 Console.WriteLine("insert password");
447 string password = Console.ReadLine();
448 var credential = new PasswordCredential() { UserName = username, Password = password };
449 if (credential != null)
450 {
451 args.AcceptWithPasswordCredential(credential);
452 }
453 collectCredentialDeferral.Complete();
454
455 break;
456
457 case DevicePairingKinds.ConfirmPinMatch:
458 // We show the PIN here and the user responds with whether the PIN matches what they see
459 // on the target device. Response comes back and we set it on the PinComparePairingRequestedData
460 // then complete the deferral.
461 var displayMessageDeferral = args.GetDeferral();
462 Console.WriteLine("pin: " + args.Pin);
463 Console.WriteLine("does the pin matches? Y/N");
464 string answer = Console.ReadLine();
465 while (!answer.ToLower().Equals("y") && !answer.ToLower().Equals("n"))
466 {
467 Console.WriteLine("please answer y or n");
468 answer = Console.ReadLine();
469 }
470 if (answer.ToLower().Equals("y"))
471 {
472 args.Accept();
473 }
474
475
476 displayMessageDeferral.Complete();
477 break;
478 }

492 public static async Task<PairingResult> PairDevice(ICosmedBleDevice device, DevicePairingKinds ceremonySelection, DevicePairingProtectionLevel minProtectionLevel)
493 {
494 //saves the device information
495 DeviceInformation deviceInformation = device.BluetoothLeDevice.DeviceInformation;
496 try
497 {
498 //create a new device from deviceId (necessary to recognize previously paired devices)
499 var bledevice = device.BluetoothLeDevice; // await BluetoothLEDevice.FromIdAsync(deviceInformation.Id);
500
501 //try the pairing process
502 bledevice.DeviceInformation.Pairing.Custom.PairingRequested += PairingRequestedHandler;
503 DevicePairingResult devicePairingResult = await bledevice.DeviceInformation.Pairing.Custom.PairAsync(ceremonySelection, minProtectionLevel);
504 bledevice.DeviceInformation.Pairing.Custom.PairingRequested -= PairingRequestedHandler;
505
506 string deviceId = deviceInformation.Id;
507
508 //update the device
509 await device.UpdateDeviceAsync(deviceId);
510
511 //bledevice = await BluetoothLEDevice.FromIdAsync(deviceId);
513 //device.bluetoothLeDevice = bledevice;
514
515 return new PairingResult(devicePairingResult.ProtectionLevelUsed, devicePairingResult.Status, bledevice.WasSecureConnectionUsedForPairing);
516 }
517 catch (Exception e)
518 {
519 //system exceptions
520 throw;
521 }
522 }
