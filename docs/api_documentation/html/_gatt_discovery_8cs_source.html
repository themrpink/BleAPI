274 public async Task<GattDeviceServicesResult> FindGattServicesByUuidAsync(Guid requestedUuid, BluetoothCacheMode cacheMode = BluetoothCacheMode.Uncached)
275 {
276 try
277 {
278 GattDeviceServicesResult services = await Device.BluetoothLeDevice.GetGattServicesForUuidAsync(requestedUuid, cacheMode).AsTask();

400 public async Task<GattDeviceServicesResult> GetAllGattServicesAsync(BluetoothCacheMode bluetoothCacheMode = BluetoothCacheMode.Uncached)
401 {
402 var accessStatus = await Device.BluetoothLeDevice.RequestAccessAsync();
403 if (accessStatus == DeviceAccessStatus.Allowed)
404 {
405 try
406 {
407 //how to set cache mode:
408 //https://docs.microsoft.com/en-us/uwp/api/windows.devices.bluetooth.bluetoothcachemode?view=winrt-22000
409 var gattResult = await Device.BluetoothLeDevice.GetGattServicesAsync(bluetoothCacheMode).AsTask();
410 gattResults.Add(gattResult);
411 return gattResult;
412 }
413 catch (Exception e)
414 {
415 //Console.WriteLine("ERROR: " + e.Message + " --- device name: " + Device.BluetoothLeDevice.Name);
416 throw new GattCommunicationException("impossible to retrieve the services", e);
417 }
418 }
419 return null;

441 public void ClearServices()
442 {
443 //Device.BluetoothLeDevice.GattServicesChanged -= GattServicesChangedHandler;
444 //GattSession.SessionStatusChanged -= SessionStatusChangedHandler;
445 //GattSession.MaxPduSizeChanged -= MaxPduSizeChangedHandler;
446
447 GattSession.Dispose();
448 GattSession = null;
449
450 foreach (var gattResult in gattResults)
451 {
452 if (gattResult != null)
453 {
454 foreach (var service in gattResult.Services)
455 {
456 if (service != null)
457 {
458 service.Dispose();
459 }
460 }
461 }
462 }
463
464 gattResults = null;
465
466 GC.Collect();
467 }
